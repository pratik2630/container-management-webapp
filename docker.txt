Docker commands to be inserted.

#...Pull or download images/OS...

docker pull <img-name:version>


#...Docker run command...

docker run -i -t --name container-name image-name


#...To start stoppes container/OS...

docker start <img-name>


#...To check status of docker service...

systemctl status docker


#...To start docker service...

systemctl start docker


#...To check list of docker images...

docker images


#...To get all information about docker...

docker info


#...To check running OS...
 
docker ps


# to check all installed images and check info about all(running/stopped)...

docker ps -a


#...To stop launched OS/Img...

docker stop


#...For make any service permanent available...

systemctl enable docker


#...To attach/conect with OS...

docker attach <os-id/img-name>


#...To remove image/OS/Container permanently(This cmd not remove running OS)...

docker rm <container-id/img-name>


#...To remove image using rmi command (rmi- remove image)...

docker rmi <img-name:tag>


#...To remmove all containers permanently included running OS...

docker rm -f $(docker ps -a -q)


#...To print ID of all available OS/Images...

docker ps -a -q


#...Create a copy of the old image with new name...

docker tag <old-name> <new-name>


#...Rename container...

docker rename <my_container> <my_new_container>


#...To push or to share your images to the Docker Hub registry or to a self-hosted one...

docker push <OPTIONS> <NAME:TAG>


#...To restart one or more containers...

docker restart <my_container>


#...To Search Docker Hub for images...

docker search <TERM>


#...To Manage Docker secrets...

docker secret <COMMAND>


#...To Creates a secret using standard input or from a file for the secret content...

 docker secret create <my_secret>


#...To Display detailed information on one or more secrets...

docker secret inspect <my-secret>


#...list of secrets...

docker secret ls


#...To Remove one or more secrets...

docker secret rm <my-secret>
